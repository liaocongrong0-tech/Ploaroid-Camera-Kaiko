// Audio
const AudioCtx = window.AudioContext || window.webkitAudioContext;
const audioCtx = new AudioCtx();
function playShutterSound() {
  if (audioCtx.state === 'suspended') audioCtx.resume();
  const t = audioCtx.currentTime;
  const osc = audioCtx.createOscillator(), g = audioCtx.createGain();
  osc.type = 'square';
  osc.frequency.setValueAtTime(600, t);
  osc.frequency.exponentialRampToValueAtTime(80, t + 0.08);
  g.gain.setValueAtTime(0.7, t); g.gain.exponentialRampToValueAtTime(0.001, t + 0.08);
  osc.connect(g); g.connect(audioCtx.destination); osc.start(); osc.stop(t + 0.08);
  const bl = audioCtx.sampleRate * 1.2;
  const buf = audioCtx.createBuffer(1, bl, audioCtx.sampleRate);
  const d = buf.getChannelData(0); for(let i=0;i<bl;i++) d[i]=Math.random()*2-1;
  const src = audioCtx.createBufferSource(); src.buffer = buf;
  const flt = audioCtx.createBiquadFilter(); flt.type = 'lowpass'; flt.frequency.value = 350;
  const ng = audioCtx.createGain();
  ng.gain.setValueAtTime(0.15, t+0.08); ng.gain.linearRampToValueAtTime(0, t+1.3);
  src.connect(flt); flt.connect(ng); ng.connect(audioCtx.destination); src.start(t+0.08);
}

async function downloadLayout() {
  const W = window.innerWidth, H = window.innerHeight, SC = 2;
  const oc2 = document.createElement('canvas');
  oc2.width = W*SC; oc2.height = H*SC;
  const oc = oc2.getContext('2d');
  oc.scale(SC, SC);

  // Background
  const bc = document.body.className;
  if (bc === 'bg-night') { oc.fillStyle = '#1a1614'; }
  else if (bc === 'bg-sage')  { oc.fillStyle = '#c8cfc4'; }
  else if (bc === 'bg-blush') { oc.fillStyle = '#e2c4c4'; }
  else if (bc === 'bg-gradient') {
    const g = oc.createLinearGradient(0,0,W,H);
    g.addColorStop(0,'#ffd6cc'); g.addColorStop(.4,'#fce4ec'); g.addColorStop(1,'#e8d5f0');
    oc.fillStyle = g;
  } else { oc.fillStyle = '#f5f2ee'; }
  oc.fillRect(0,0,W,H);

  // Collect items sorted by z-index
  const items = [];
  document.querySelectorAll('.polaroid-photo').forEach(el => items.push({type:'photo',el,z:+el.style.zIndex||0}));
  document.querySelectorAll('.board-sticker').forEach(el  => items.push({type:'sticker',el,z:+el.style.zIndex||0}));
  items.push({ type:'camera', el: document.getElementById('camera-wrapper'), z:100 });
  items.sort((a,b) => a.z - b.z);

  for (const item of items) {
    const rect = item.el.getBoundingClientRect();
    if (item.type === 'photo')   await renderPhoto(oc, item.el, rect);
    if (item.type === 'sticker') await renderSticker(oc, item.el, rect);
    if (item.type === 'camera')  renderCamera(oc, rect);
  }

  const a = document.createElement('a');
  a.download = `Polaroid_${Date.now()}.png`;
  a.href = oc2.toDataURL('image/png');
  a.click();
}

function rrect(c, x,y,w,h,r) {
  if (typeof r === 'number') r=[r,r,r,r];
  c.beginPath();
  c.moveTo(x+r[0],y); c.lineTo(x+w-r[1],y); c.arcTo(x+w,y,x+w,y+r[1],r[1]);
  c.lineTo(x+w,y+h-r[2]); c.arcTo(x+w,y+h,x+w-r[2],y+h,r[2]);
  c.lineTo(x+r[3],y+h); c.arcTo(x,y+h,x,y+h-r[3],r[3]);
  c.lineTo(x,y+r[0]); c.arcTo(x,y,x+r[0],y,r[0]); c.closePath();
}

      const cls = [...imgEl.classList];
      let f = 'sepia(0.28) contrast(1.12) brightness(0.97) saturate(0.88)';
      if(cls.includes('film-fuji'))    f='hue-rotate(8deg) saturate(1.1) contrast(1.08) brightness(0.93) sepia(0.15)';
      if(cls.includes('film-ilford'))  f='grayscale(0.85) contrast(1.2) brightness(0.94) sepia(0.12)';
      if(cls.includes('film-lomo'))    f='saturate(1.35) contrast(1.3) brightness(0.88) hue-rotate(-12deg) sepia(0.1)';
      if(cls.includes('film-expired')) f='sepia(0.55) contrast(0.9) brightness(0.88) saturate(0.6) hue-rotate(5deg)';
      oc.filter = f;
      oc.drawImage(tmp,0,0,pw,ph);
      oc.filter = 'none';
      // Vignette
      const vg = oc.createRadialGradient(pw/2,ph/2,pw*.28,pw/2,ph/2,pw*.75);
      vg.addColorStop(0,'transparent'); vg.addColorStop(1,'rgba(0,0,0,0.40)');
      oc.fillStyle=vg; oc.fillRect(0,0,pw,ph);
      oc.restore();
      // Labels
      oc.fillStyle='#bbb'; oc.font='7.5px DM Mono,monospace'; oc.textAlign='right';
      const lbl=el.querySelector('.film-label');
      if(lbl) oc.fillText(lbl.textContent, rect.width/2-12, rect.height/2-12);
      const ts=el.querySelector('.photo-timestamp');
      if(ts){ oc.textAlign='left'; oc.fillText(ts.textContent,-rect.width/2+12,rect.height/2-12); }
      oc.restore(); resolve();
    };
    tmp.onerror=()=>{ oc.restore(); resolve(); };
  });
}

async function renderSticker(oc, el, rect) {
  const svgEl = el.querySelector('svg'); if(!svgEl) return;
  const clone = svgEl.cloneNode(true);
  clone.setAttribute('width','80'); clone.setAttribute('height','80');
  const blob = new Blob([new XMLSerializer().serializeToString(clone)],{type:'image/svg+xml'});
  const url  = URL.createObjectURL(blob);
  return new Promise(resolve => {
    const img = new Image();
    img.onload = () => {
      const rotM = el.style.transform.match(/rotate\(([^)]+)deg\)/);
      const ang  = rotM ? parseFloat(rotM[1]) * Math.PI/180 : 0;
      const cx = rect.left+rect.width/2, cy = rect.top+rect.height/2;
      oc.save(); oc.translate(cx,cy); oc.rotate(ang);
      oc.drawImage(img,-rect.width/2,-rect.height/2,rect.width,rect.height);
      oc.restore(); URL.revokeObjectURL(url); resolve();
    };
    img.onerror=()=>{ URL.revokeObjectURL(url); resolve(); };
    img.src=url;
  });
}

// Drag
function makeDraggable(el) {
  let p1=0,p2=0,p3=0,p4=0;
  el.addEventListener('mousedown', s);
  el.addEventListener('touchstart', e=>s(e.touches[0]),{passive:true});
  function s(e) {
    if(e.target.classList.contains('delete-btn')||e.target.classList.contains('sticker-delete')) return;
    el.style.zIndex=zBase++;
    p3=e.clientX; p4=e.clientY;
    document.addEventListener('mousemove',m); document.addEventListener('mouseup',u);
    document.addEventListener('touchmove',t=>m(t.touches[0]),{passive:true}); document.addEventListener('touchend',u);
  }
  function m(e) {
    p1=p3-e.clientX; p2=p4-e.clientY; p3=e.clientX; p4=e.clientY;
    el.style.top=(el.offsetTop-p2)+'px'; el.style.left=(el.offsetLeft-p1)+'px';
  }
  function u() {
    document.removeEventListener('mousemove',m); document.removeEventListener('mouseup',u);
    document.removeEventListener('touchmove',m); document.removeEventListener('touchend',u);
  }
}
